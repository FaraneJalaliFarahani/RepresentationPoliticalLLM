  - `scipy==1.12.0`
  - `tqdm==4.66.2`
  - `transformers==4.38.2`
  - `matplotlib==3.8.3`
  - `numpy==1.26.4`
  - `pandas==2.2.1`
  - `pyvene==0.1.1`
  - `torch==2.3.0`
  - `seaborn==0.13.2`
